employees
-
emp_no PK int
birth_date date
first_name varchar
last_name varchar
gender char(1)
hire_date date

salaries
-
emp_no PK int FK >- employees.emp_no
salary int
from_date PK date
to_date date

titles
-
emp_no PK int FK >- employees.emp_no
title varchar
from_date PK date
to_date date

departments
-
dept_no PK varchar(4)
dept_name varchar UNIQUE

dept_emp
-
emp_no PK int FK >- employees.emp_no
dept_no PK varchar(4) FK >- departments.dept_no
from_date date
to_date date

dept_manager
-
dept_no PK varchar(4) FK >- departments.dept_no
emp_no int FK >- employees.emp_no
from_date PK date
to_date date
